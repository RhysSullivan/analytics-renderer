# Querying Post Hog

### Supported:

- Trends

### Unsupported:

- Funnels
- Retention
- User Paths
- Stickiness
- Lifecycle
- SQL / HogQL

```ts
const data = await posthog
  .query()
  .addSeries({
    name: "$pageview",
    sampling: "total",
    where: [
      {
        compare: "equals",
        name: "$browser",
        value: "Chrome",
      },
      {
        compare: "equals",
        name: "$os",
        value: "Windows",
      },
    ],
  })
  .addSeries({
    sampling: "total",
    name: "Asked Question",
  })
  .addFilterGroup({
    match: "all",
    filters: {
      compare: "equals",
      name: "$geoip_country_code",
      value: "US",
    },
  })
  .addFilterGroup({
    match: "all",
    filters: {
      compare: "equals",
      name: "$pathname",
      value: "/about",
    },
  })
  .execute({
    groupBy: "day",
    type: "line",
  });
```
